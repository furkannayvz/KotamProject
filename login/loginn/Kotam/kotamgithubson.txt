    final url = Uri.parse('http://34.14.39.115/api/packages/all');
    try {
      final response = await http.get(url);
      if (response.statusCode == 200) {
        setState(() {
          tariffs = json.decode(response.body);
        });
      }
    } catch (e) {
      print('Hata: $e');
    }

    Navigator.pushNamedAndRemoveUntil(context, '/', (_) => false);
